# logfile_logfile_reset logfile_reset () {

local res=0

if [ x$logfile_status == x”1” ] ; then

confirmation_question “Are you sure to remove file $LOGFILE ? [y/N]” res=$?

if [ $res -eq 0 ] ; then
rm $LOGFILE || error_handled “Error on remove file $LOGFILE” echo -en “File $LOGFILE removed correctly.n”
else
res=1

fi

else

echo -en “Logging is not enable. I do nothing.n” res=1

fi

return $res

} # logfile_logfile_reset_end
