# psql_set_auth_var psql_set_auth_var () {

local db=$1 local user=$2 local pwd=$3 local host=$4 local schema=$5 local v_host=”” local v_schema=”“

if [ ! -z “$host” ] ; then
v_host=”-h $host”

fi

if [ ! -z “$schema” ] ; then
v_schema=”-v schema=${schema}”

fi

export PGPASSWORD=”$pwd” # TODO: handle ~/.pgpass file

psql_auth=”$v_host $v_schema -U $user -w $db”

[[ $DEBUG && $DEBUG == true ]] && echo “Use ‘$psql_auth’”

return 0

} # psql_set_auth_var_end
